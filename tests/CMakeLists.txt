macro(add_dispar_test name)
  set(targetName "test_${name}")
  add_dispar_executable(
    ${targetName}

    main.cc

    testutils.h
    testutils.cc

    binaries.qrc

    ${ARGN}
    )

  add_dependencies(tests ${targetName})

  add_test(
    NAME ${name}
    COMMAND ${targetName}
    )

  target_link_libraries(
    ${targetName}
    gtest
    ${libName}
    )

  if (LINUX)
    target_link_libraries(
      ${targetName}
      pthread
      )
  endif()

  include_directories(
    gtest/include
    ../src
    )

  qt5_use_modules(
    ${targetName}
    Core
    Test
    )
endmacro()
